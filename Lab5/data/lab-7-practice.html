def get_critical_value(size, alpha):
    t_dist = stats.t.ppf(1 - alpha / (2 * size), size - 2)
    numerator = (size - 1) * np.sqrt(np.square(t_dist))
    denominator = np.sqrt(size) * np.sqrt(size - 2 + np.square(t_dist))
    critical_value = numerator / denominator
    return critical_value

def grubbs_test(data, alpha=0.05):
    g_calculated = np.abs(data - np.mean(data)).max() / np.std(data)
    g_critical = get_critical_value(len(data), alpha)
    print(f'  : {g_calculated},  : {g_critical}')
    return g_calculated, g_critical

def calc_valid_range(data):
    complete_cases = data[~data.isna()]
    q25, q75 = complete_cases.quantile([0.25, 0.75])['CO2 emission']
    qr = q75 - q25
    return (q25 - 1.5 * qr, q75 + 1.5 * qr)

def lognorm(data):
    mean = np.mean(data)
    std = np.std(data)
    norm_data = 1 / (1 + np.exp(mean - data) / (std + 0.0000001))
    return norm_data
